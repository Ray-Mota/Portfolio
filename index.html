// busca endereço por CEP
async function consultarCEP(cep) {
const url = `viacep.com/ws/${cep}/json`;
const res = await fetch(url);
const data = await res.json();
return {
rua: data.logradouro,
cidade: data.localidade,
estado: data.uf
}
}
